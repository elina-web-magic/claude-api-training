### 2.1 Rate Limiting

```typescript
// lib/security/rate-limiter.ts

import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";
import prisma from "@/lib/prisma";

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

export interface RateLimitResult {
  success: boolean;
  limit: number;
  remaining: number;
  reset: number;
  retryAfter?: number;
}

/**
 * Global rate limiter (per IP)
 */
export const globalRateLimiter = new Ratelimit({
  redis,
  limiter: Ratelimit.slidingWindow(100, "1 h"),
  analytics: true,
  prefix: "rl:global",
});

/**
 * Per-API-Key rate limiter
 */
export async function checkApiKeyRateLimit(
  apiKeyId: string
): Promise<RateLimitResult> {
  try {
    // Get rate limit config for this API key
    const config = await prisma.rateLimitConfig.findUnique({
      where: { apiKeyId },
    });

    if (!config) {
      return {
        success: false,
        limit: 0,
        remaining: 0,
        reset: 0,
      };
    }

    // Check minute window
    const now = new Date();
    const minuteAgo = new Date(now.getTime() - 60 * 1000);

    const minuteLogs = await prisma.rateLimitLog.findMany({
      where: {
        apiKeyId,
        windowStart: {
          gte: minuteAgo,
        },
      },
    });

    const minuteCount = minuteLogs.reduce((sum, log) => sum + log.requestCount, 0);

    if (minuteCount >= config.requestsPerMinute) {
      return {
        success: false,
        limit: config.requestsPerMinute,
        remaining: 0,
        reset: Math.ceil((minuteLogs[0].windowEnd.getTime() - now.getTime()) / 1000),
        retryAfter: Math.ceil((minuteLogs[0].windowEnd.getTime() - now.getTime()) / 1000),
      };
    }

    // Log this request
    await prisma.rateLimitLog.upsert({
      where: {
        apiKeyId_endpoint_windowStart: {
          apiKeyId,
          endpoint: "generic",
          windowStart: new Date(Math.floor(now.getTime() / 60000) * 60000),
        },
      },
      update: {
        requestCount: {
          increment: 1,
        },
      },
      create: {
        apiKeyId,
        endpoint: "generic",
        requestCount: 1,
        windowStart: new Date(Math.floor(now.getTime() / 60000) * 60000),
        windowEnd: new Date(Math.floor(now.getTime() / 60000) * 60000 + 60000),
      },
    });

    return {
      success: true,
      limit: config.requestsPerMinute,
      remaining: Math.max(0, config.requestsPerMinute - minuteCount - 1),
      reset: Math.ceil((new Date(Math.floor(now.getTime() / 60000) * 60000 + 60000).getTime() - now.getTime()) / 1000),
    };
  } catch (error) {
    console.error("Rate limit check error:", error);
    // Fail open to avoid blocking legitimate requests
    return {
      success: true,
      limit: 0,
      remaining: 0,
      reset: 0,
    };
  }
}

/**
 * IP-based rate limiting for public endpoints
 */
export async function checkIPRateLimit(
  ip: string
): Promise<RateLimitResult> {
  try {
    const ipRateLimiter = new Ratelimit({
      redis,
      limiter: Ratelimit.slidingWindow(30, "1 h"),
      prefix: `rl:ip:${ip}`,
    });

    const { success, limit, remaining, reset, retryAfter } = await ipRateLimiter.limit(ip);

    return {
      success,
      limit,
      remaining,
      reset,
      retryAfter,
    };
  } catch (error) {
    console.error("IP rate limit check error:", error);
    return {
      success: true,
      limit: 0,
      remaining: 0,
      reset: 0,
    };
  }

```typescript
// lib/arxiv/fetcher.ts

import { parseStringPromise } from "xml2js";
import prisma from "@/lib/prisma";
import { logger } from "@/lib/monitoring/logger";
import * as Sentry from "@sentry/nextjs";

export interface ArxivArticleRaw {
  id: string;
  title: string;
  summary: string;
  authors: Array<{
    name: string;
  }>;
  published: string;
  updated: string;
  link: string;
  category: string;
}

export interface ProcessedArticle {
  title: string;
  slug: string;
  description: string;
  content: string;
  insights: string;
  theme_area: string;
  mermaidSchema: string;
  tags: string[];
  category: string;
  metaTitle: string;
  metaDescription: string;
  metaKeywords: string;
  coverImage?: string;
  coverImageAlt?: string;
  arxivId?: string;
  arxivUrl?: string;
}

const ARXIV_RSS_URL = "https://arxiv.org/list/cs.AI/";
const ARXIV_FEED_URL = "https://arxiv.org/rss/cs.AI";

/**
 * Fetch and parse ArXiv RSS feed
 */
export async function fetchArxivFeed(): Promise<ArxivArticleRaw[]> {
  try {
    logger.info("Fetching ArXiv AI feed", { url: ARXIV_FEED_URL });

    const response = await fetch(ARXIV_FEED_URL, {
      headers: {
        "User-Agent": "AILens-Bot/1.0 (+https://ailens.app)",
      },
      timeout: 30000,
    });

    if (!response.ok) {
      throw new Error(`ArXiv feed fetch failed: ${response.status} ${response.statusText}`);
    }

    const feedContent = await response.text();
    const parsedFeed = await parseStringPromise(feedContent);

    const articles: ArxivArticleRaw[] = [];
    const feedItems = parsedFeed.feed?.entry || [];

    for (const item of feedItems) {
      try {

/**
 * Parse individual ArXiv RSS entry
 */
function parseArxivEntry(entry: any): ArxivArticleRaw | null {
  try {
    const id = entry.id?.[0] || "";
    const title = entry.title?.[0]?.trim() || "";
    const summary = entry.summary?.[0]?.trim() || "";
    const published = entry.published?.[0] || "";
    const updated = entry.updated?.[0] || "";

    // Extract arXiv ID from entry ID
    // Format: http://arxiv.org/abs/2024.01234v1
    const arxivIdMatch = id.match(/\/(\d+\.\d+)/);
    const arxivId = arxivIdMatch ? arxivIdMatch[1] : "";

    // Parse authors
    const authorArray = Array.isArray(entry.author) ? entry.author : [];
    const authors = authorArray.map((author: any) => ({
      name: author.name?.[0] || "Unknown",
    }));

    // Get primary category
    const category = entry["arxiv:primary-category"]?.[0]?.$.term || "cs.AI";

    // Get link
    const links = Array.isArray(entry.link) ? entry.link : [];
    const link =
      links.find((l: any) => l.$.type === "text/html")?.$.href ||
      links[0]?.$.href ||
      `https://arxiv.org/abs/${arxivId}`;

    if (!title || !summary) {
      logger.warn("ArXiv entry missing required fields", {
        arxivId,
        hasTitle: !!title,
        hasSummary: !!summary,
      });
      return null;
    }

    return {
      id: arxivId,
      title,
      summary,
      authors,
      published,
      updated,
      link,
      category,
    };
  } catch (error) {
    logger.warn("Error parsing ArXiv entry", {
      error: error instanceof Error ? error.message : String(error),
    });
    return null;
  }
}

/**
 * Sleep utility for exponential backoff
 */
function sleep(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

/**
 * Calculate exponential backoff delay with jitter
 */
function calculateBackoffDelay(
  attempt: number,
  config: RetryConfig
): number {
  const exponentialDelay = Math.min(
    config.initialDelayMs * Math.pow(config.backoffMultiplier, attempt),
    config.maxDelayMs
  );

  // Add jitter: random value between 0 and exponentialDelay
  const jitter = Math.random() * exponentialDelay;
  return exponentialDelay + jitter;
}

/**
 * Determine if error is retryable
 */
function isRetryableError(error: unknown): boolean {
  // Rate limit errors
  if (error instanceof Anthropic.RateLimitError) {
    logger.warn("Rate limit error detected, will retry", {
      status: error.status,
      message: error.message,
    });
    return true;
  }

  // Timeout/Network errors
  if (error instanceof Anthropic.APIConnectionError) {
    logger.warn("Connection error detected, will retry", {
      message: error.message,
    });
    return true;
  }

  // Temporary server errors (5xx)
  if (error instanceof Anthropic.APIStatusError) {
    const status = error.status;
    if (status >= 500 && status < 600) {
      logger.warn("Server error detected, will retry", {
        status,
        message: error.message,
      });
      return true;
    }

    // 408 Request Timeout
    if (status === 408) {
      logger.warn("Request timeout, will retry", {
        status,
        message: error.message,
      });
      return true;
    }
  }

  return false;
}

 * Extract article data with Claude API
 * Implements exponential backoff retry logic
 * Assumes Claude returns PERFECT JSON - no string manipulation
 */
export async function extractArticleDataWithRetry(
  articleTitle: string,
  articleSummary: string,
  arxivCategory: string,
  retryConfig: RetryConfig = DEFAULT_RETRY_CONFIG
): Promise<ClaudeExtractionResult> {
  let lastError: Error | null = null;

  for (let attempt = 0; attempt <= retryConfig.maxRetries; attempt++) {
    try {
      logger.info("Calling Claude API for article extraction", {
        attempt: attempt + 1,
        maxRetries: retryConfig.maxRetries + 1,
        title: articleTitle.substring(0, 50),
      });

      const result = await callClaudeAPI(
        articleTitle,
        articleSummary,
        arxivCategory
      );

      logger.info("Successfully extracted article data from Claude", {
        attempt: attempt + 1,
        themeArea: result.themeArea,
        hashtagCount: result.hashtags.length,
      });

      return result;
    } catch (error) {
      lastError = error instanceof Error ? error : new Error(String(error));

      const isRetryable = isRetryableError(error);

      if (!isRetryable) {
        logger.error("Non-retryable error from Claude API", {
          error: lastError.message,
          attempt: attempt + 1,
        });

        Sentry.captureException(lastError, {
          tags: {
            source: "claude-client",
            operation: "extractArticleDataWithRetry",
            retryable: false,
          },
          extra: {
            attempt: attempt + 1,
            title: articleTitle.substring(0, 100),
          },
        });

        throw lastError;
      }

      // Check if we should retry
      if (attempt < retryConfig.maxRetries) {
        const delayMs = calculateBackoffDelay(attempt, retryConfig);

        logger.warn("Retryable error detected, will retry with backoff", {
          error: lastError.message,
          attempt: attempt + 1,
          nextRetryDelayMs: delayMs,
          maxRetries: retryConfig.maxRetries,
        });

        await sleep(delayMs);
      }
    }
  }

  // All retries exhausted
  logger.error("All Claude API retry attempts exhausted", {
    maxRetries: retryConfig.maxRetries,
    lastError: lastError?.message,
    title: articleTitle.substring(0, 50),
  });

  Sentry.captureException(lastError, {
    tags: {
      source: "claude-client",
      operation: "extractArticleDataWithRetry",
      retryable: true,
      retriesExhausted: true,
    },
    extra: {
      totalAttempts: retryConfig.maxRetries + 1,
      title: articleTitle.substring(0, 100),
    },
  });

  throw new Error(
    `Failed to extract article data after ${retryConfig.maxRetries + 1} attempts: ${lastError?.message}`
  );
}

/**
 * Validate basic format requirements
 */
function validateBasicFormat(schema: string): MermaidValidationError[] {
  const errors: MermaidValidationError[] = [];

  // Check if empty
  if (!schema || schema.trim().length === 0) {
    errors.push({
      line: 1,
      column: 0,
      code: "EMPTY_SCHEMA",
      message: "Mermaid schema is empty",
      snippet: "",
    });
    return errors;
  }

  // Check if too long (arbitrary but reasonable limit)
  if (schema.length > 5000) {
    errors.push({
      line: 1,
      column: 0,
      code: "SCHEMA_TOO_LONG",
      message: `Mermaid schema exceeds maximum length (${schema.length} > 5000)`,
      snippet: schema.substring(0, 100),
    });
  }

  // Check for balanced brackets
  let squareBracketCount = 0;
  let curlyBracketCount = 0;
  let roundBracketCount = 0;

  for (let i = 0; i < schema.length; i++) {
    const char = schema[i];

    if (char === "[") squareBracketCount++;
    if (char === "]") squareBracketCount--;
    if (char === "{") curlyBracketCount++;
    if (char === "}") curlyBracketCount--;
    if (char === "(") roundBracketCount++;
    if (char === ")") roundBracketCount--;

    if (squareBracketCount < 0 || curlyBracketCount < 0 || roundBracketCount < 0) {
      const lineNumber = schema.substring(0, i).split("\n").length;
      errors.push({
        line: lineNumber,
        column: i,
        code: "UNBALANCED_BRACKETS",
        message: "Unbalanced brackets detected",
        snippet: schema.split("\n")[lineNumber - 1] || "",
      });
      break;
    }
  }

  if (squareBracketCount !== 0) {
    errors.push({
      line: 1,
      column: 0,
      code: "UNBALANCED_BRACKETS",
      message: `Unbalanced square brackets: ${squareBracketCount > 0 ? "missing" : "extra"} ]`,
      snippet: schema.substring(0, 100),
    });
  }

  if (curlyBracketCount !== 0) {
    errors.push({
      line: 1,
      column: 0,
      code: "UNBALANCED_BRACKETS",
      message: `Unbalanced curly brackets: ${curlyBracketCount > 0 ? "missing" : "extra"} }`,
      snippet: schema.substring(0, 100),
    });
  }

  if (roundBracketCount !== 0) {
    errors.push({
      line: 1,
      column: 0,
      code: "UNBALANCED_BRACKETS",
      message: `Unbalanced round brackets: ${roundBracketCount > 0 ? "missing" : "extra"} )`,
      snippet: schema.substring(0, 100),
    });
  }

  return errors;
}

/**
 * Parse graph declaration line
 * Valid: graph TD, graph LR, graph TB, graph RL
 */
function parseGraphDeclaration(schema: string): string | null {
  const firstLine = schema.split("\n")[0].trim();

  // Strict regex for graph declaration
  const graphRegex = /^graph\s+(TD|LR|TB|RL|BT)(\s|$)/i;
  const match = firstLine.match(graphRegex);

  if (match) {
    return match[1].toUpperCase();
  }

  return null;
}

/**
 * Parse a connection line
 * Valid formats:
 * - A --> B
 * - A -->|label| B
 * - A --- B
 * - A -. B
 * - A ==> B
 */
function parseConnection(line: string, lineNumber: number): ParsedConnection | null {
  // Connection regex - very strict
  // Format: NodeID(Arrow|Connection)NodeID or NodeID(Arrow|Connection)|Label|NodeID
  const connectionRegex = /^([A-Za-z0-9_]+)\s*(-->|---|\.->|==>|\.\.>)\s*(?:\|([^|]*)\|)?([A-Za-z0-9_]+)$/;

  const match = line.match(connectionRegex);
  if (!match) {
    return null;
  }

  const [, source, connectionType, label, target] = match;

  // Validate node IDs are alphanumeric and underscore only
  if (!/^[A-Za-z0-9_]+$/.test(source) || !/^[A-Za-z0-9_]+$/.test(target)) {
    return null;
  }

  let type: "arrow" | "solid" | "dotted" | "thick" = "solid";
  if (connectionType === "-->") type = "arrow";
  if (connectionType === "---") type = "solid";
  if (connectionType === ".->") type = "dotted";
  if (connectionType === "==>") type = "thick";

  return {
    source,
    target,
    label: label?.trim() || undefined,
    type,
    lineNumber,
  };
}

/**
 * Parse a node declaration line
 * Valid formats:
 * - A["Label"]
 * - A("Rounded")
 * - A{Diamond}
 * - A((Circle))
 * - A[/Parallelogram/]
 * - A[\\Trapezoid\\]
 * - A@{ hexagon }
 */
function parseNode(line: string, lineNumber: number): ParsedNode | null {
  // Node declaration regex - strict
  const nodeRegexes = [
    // Rectangular: A["Label"]
    {
      regex: /^([A-Za-z0-9_]+)\s*\[\s*"([^"]*)"\s*\]$/,
      type: "rectangular" as const,
    },
    // Rounded: A("Label")
    {
      regex: /^([A-Za-z0-9_]+)\s*\(\s*"([^"]*)"\s*\)$/,
      type: "rounded" as const,
    },
    // Diamond: A{Label}
    {
      regex: /^([A-Za-z0-9_]+)\s*\{\s*([^}]*)\s*\}$/,
      type: "diamond" as const,
    },
    // Circle: A((Label))
    {
      regex: /^([A-Za-z0-9_]+)\s*\(\(\s*([^)]*)\s*\)\)$/,
      type: "circle" as const,
    },
    // Hexagon: A@{ Label }
    {
      regex: /^([A-Za-z0-9_]+)\s*@\{\s*([^}]*)\s*\}$/,
      type: "hexagon" as const,
    },
    // Parallelogram: A[/Label/]
    {
      regex: /^([A-Za-z0-9_]+)\s*\[\s*\/([^/]*)\s*\/\s*\]$/,
      type: "parallelogram" as const,
    },
    // Trapezoid: A[\\Label\\]
    {
      regex: /^([A-Za-z0-9_]+)\s*\[\s*\\\\([^\\]*)\\\\\s*\]$/,
      type: "trapezoid" as const,
    },
    // Text (no formatting): A[Label]
    {
      regex: /^([A-Za-z0-9_]+)\s*\[\s*([^\]]*)\s*\]$/,
      type: "text" as const,
    },
  ];

  for (const { regex, type } of nodeRegexes) {
    const match = line.match(regex);

export async function POST(request: NextRequest): Promise<NextResponse<CronJobResponse>> {

              code: "AUTHENTICATION_FAILED",
              message: authError,
            },
          ],

        { status: 401 }

        status: "PROCESSING",

      jobId,

    } catch (error) {
      const errorMsg = error instanceof Error ? error.message : String(error);
      logger.error("Failed to fetch ArXiv feed", {
        jobId,
        error: errorMsg,
      });

        code: "ARXIV_FETCH_ERROR",
        message: `Failed to fetch ArXiv feed: ${errorMsg}`,
      });

      // Update cron job as failed
      await prisma.cronJobRun.update({
        where: { id: cronJobRun.id },
        data: {
          status: "FAILED",
          errorMessage: errorMsg,
          completedAt: new Date(),
        },
      });

          success: false,
          jobId,
          timestamp: new Date().toISOString(),

/**
 * Validate Bearer token from Authorization header
 * CRITICAL SECURITY: Strict token validation
 */
function validateBearerToken(request: NextRequest): string | null {
  try {
    // Check if CRON_SECRET_KEY is configured
    if (!CRON_SECRET_KEY) {
      logger.error("CRON_SECRET_KEY not configured");
      return "Server configuration error: CRON_SECRET_KEY not set";
    }

    // Get Authorization header
    const authHeader = request.headers.get("authorization");
    if (!authHeader) {
      return "Missing Authorization header";
    }

    // Validate Bearer format
    const parts = authHeader.split(" ");
    if (parts.length !== 2) {
      return "Invalid Authorization header format. Expected: Bearer <token>";
    }

    const [scheme, token] = parts;

    if (scheme.toLowerCase() !== "bearer") {
      return "Invalid authentication scheme. Expected: Bearer";
    }

    if (!token || token.length === 0) {
      return "Empty bearer token";
    }

    // Validate token format (must be hex or alphanumeric)
    if (!/^[a-zA-Z0-9\-_]+$/.test(token)) {
      return "Invalid token format";
    }

    // Constant-time comparison to prevent timing attacks
    if (!constantTimeCompare(token, CRON_SECRET_KEY)) {
      logger.warn("Cron job authentication failed: invalid token", {
        tokenLength: token.length,
      });
      return "Invalid authentication token";
    }

    return null; // Authentication successful
  } catch (error) {
    logger.error("Error during Bearer token validation", {
      error: error instanceof Error ? error.message : String(error),
    });
    return "Authentication validation error";
  }
}

/**
 * Constant-time string comparison to prevent timing attacks
 */
function constantTimeCompare(a: string, b: string): boolean {
  if (a.length !== b.length) {
    return false;
  }

  let result = 0;
  for (let i = 0; i < a.length; i++) {
    result |= a.charCodeAt(i) ^ b.charCodeAt(i);
  }

  return result === 0;
}

import { NextRequest, NextResponse } from "next/server";
import { SUPPORTED_LOCALES, DEFAULT_LOCALE, normalizeLocale } from "@/lib/i18n/config";

export function middleware(request: NextRequest) {
  const pathname = request.nextUrl.pathname;

  // Check if locale is in pathname
  const pathnameHasLocale = SUPPORTED_LOCALES.some(
    (locale) => pathname.startsWith(`/${locale}/`) || pathname === `/${locale}`
  );

  if (pathnameHasLocale) {
    return NextResponse.next();
  }

  // Redirect root to default locale
  if (pathname === "/") {
    return NextResponse.redirect(new URL(`/${DEFAULT_LOCALE}`, request.url));
  }

  // Get preferred locale from Accept-Language header
  const acceptLanguage = request.headers.get("accept-language");
  const preferredLocale = acceptLanguage
    ? normalizeLocale(acceptLanguage.split(",")[0])
    : DEFAULT_LOCALE;

  return NextResponse.redirect(new URL(`/${preferredLocale}${pathname}`, request.url));
}

export const config = {
  matcher: ["/((?!api|_next/static|_next/image|favicon.ico).*)"],
};

    // Fetch featured articles from database

      where: {
        featured: true,
        publishedAt: {
          not: null,
        },
        deletedAt: null,

        title: true,
        slug: true,
        description: true,
        theme_area: true,
        viewCount: true,
